Exemplo Funcional - createElement()

Código do Exemplo - createElement()


      <div id="exampleElement"></div>
      <button onclick="addElement()">Adicionar Elemento</button>
      <script>
        function addElement() {
          var newElement = document.createElement("p");
          var text = document.createTextNode("Novo elemento criado!");
          newElement.appendChild(text);
          document.getElementById("exampleElement").appendChild(newElement);
        }
      </script>
